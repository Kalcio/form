Example: From Data

The simplest form, created from data only. Form normalization is not performed.

{
    "schema": {
        "type": "object",
        "properties": {
            "name": {
                "type": "string"
            },
            "email": {
                "type": "string",
                "format": "email",
                "maxLength": 320
            },
            "web": {
                "type": "string",
                "format": "uri",
                "maxLength": 2048,
                "minLength": 2
            },
            "age": {
                "type": "integer"
            },
            "start_date": {
                "type": "string",
                "format": "date-time",
                "maxLength": 35,
                "minLength": 16
            },
            "end_date": {
                "type": "string"
            },
            "salary": {
                "type": "number"
            },
            "require_onboarding": {
                "type": "boolean"
            }
        },
        "required": []
    },
    "uischema": {
        "type": "VerticalLayout",
        "elements": [
            {
                "type": "Control",
                "label": "Name",
                "scope": "#/properties/name"
            },
            {
                "type": "Control",
                "label": "Email",
                "scope": "#/properties/email"
            },
            {
                "type": "Control",
                "label": "Web",
                "scope": "#/properties/web"
            },
            {
                "type": "Control",
                "label": "Age",
                "scope": "#/properties/age"
            },
            {
                "type": "Control",
                "label": "Start Date",
                "scope": "#/properties/start_date"
            },
            {
                "type": "Control",
                "label": "End Date",
                "scope": "#/properties/end_date"
            },
            {
                "type": "Control",
                "label": "Salary",
                "scope": "#/properties/salary"
            },
            {
                "type": "Control",
                "label": "Require Onboarding",
                "scope": "#/properties/require_onboarding"
            }
        ]
    },
    "data": {
        "name": "John Doe",
        "email": "john.doe@example.com",
        "web": "https://www.example.com",
        "age": 39,
        "start_date": "2025-02-28T21:01:57+00:00",
        "end_date": "",
        "salary": 2499.99,
        "require_onboarding": true
    }
}
The demo in JavaScript uses the JSON Forms library.
